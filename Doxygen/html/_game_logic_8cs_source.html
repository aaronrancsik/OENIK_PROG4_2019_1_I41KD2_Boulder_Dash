65  public GameModel LoadLevel(byte[] levelResource)
66  {
67  this.originalMap = levelResource;
68  string[] lines = this.LoadFileLinesFromResource(levelResource);
69  this.width = int.Parse(lines[0]); // cella szeleseeg, magasseg
70  this.height = int.Parse(lines[1]);
71  this.model = new GameModel(this.width, this.height);
72  this.model.RequireDiamonds = int.Parse(lines[2]);
73  this.model.TextureSet = int.Parse(lines[3]);
74  for (int x = 0; x < this.width; x++)
75  {
76  for (int y = 0; y < this.height; y++)
77  {
78  char current = lines[y + 4][x];
79  var point = new Point(x, y);
80  var initPrev = new Point(x, y);
81  switch (current)
82  {
83  case 'w':
84  this.model.WallMatrix[x, y] = true;
85  break;
86  case 'X':
87  this.model.Rockford = new Rockford();
88  this.model.Rockford.TilePosition = point;
89  this.model.Rockford.TileOldPosition = initPrev;
90 
91  // model.Camera.Follow(point);
92  break;
93  case 'P':
94 
95  this.model.Exit.TilePosition = new Point(x, y);
96  break;
97  case 'r':
98  var boulder = new Blocks.Boulder();
99  boulder.TilePosition = point;
100  boulder.TileOldPosition = initPrev;
101  boulder.Variant = rnd.Next(1, 5);
102  this.model.Boulders[x, y] = boulder;
103  break;
104  case 'd':
105  var diamond = new Blocks.Diamond();
106  diamond.TilePosition = point;
107  diamond.TileOldPosition = initPrev;
108  this.model.Diamonds[x, y] = diamond;
109  break;
110  case '.':
111  var dirt = new Blocks.Dirt();
112  dirt.Variant = rnd.Next(1, 5);
113  this.model.DirtMatrix[x, y] = dirt;
114  break;
115  case 'W':
116  this.model.TitaniumMatrix[x, y] = true;
117  break;
118  case 'q':
119  this.model.Fireflies[x, y] = new Firefly(point);
120  break;
121  case 'B':
122  this.model.Butterflies[x, y] = new Butterfly(point);
123  break;
124  }
125  }
126  }
127 
128  this.CameraFollowRockford();
129  this.model.Rockford.Direaction = State.Birth;
130  return this.model;
131  }

209  private bool TryMove(Direction dir, ref int x, ref int y)
210  {
211  switch (dir)
212  {
213  case Direction.Up:
214  x += 0;
215  y += -1;
216  break;
217  case Direction.Down:
218  x += 0;
219  y += 1;
220  break;
221  case Direction.Left:
222  x += -1;
223  y += 0;
224  break;
225  case Direction.Right:
226  x += 1;
227  y += 0;
228  break;
229  }
230 
231  if (x < 0 || x >= this.model.WallMatrix.GetLength(0) || y < 0 || y >= this.model.WallMatrix.GetLength(1))
232  {
233  return false;
234  }
235 
236  if (this.model.TitaniumMatrix[x, y])
237  {
238  return false;
239  }
240 
241  if (this.model.WallMatrix[x, y])
242  {
243  return false;
244  }
245 
246  if (!this.model.Exit.IsOpen && this.model.Exit.TilePosition.Equals(new Point(x, y)))
247  {
248  return false;
249  }
250 
251  if ((dir == Direction.Up || dir == Direction.Down) && this.model.Boulders[x, y] != null)
252  {
253  return false;
254  }
255 
256  bool rollLuck = rnd.Next(100) < 30;
257  if (dir == Direction.Left)
258  {
259  if (this.model.Boulders[x, y] != null)
260  {
261  if (!this.NothingHere(x - 1, y))
262  {
263  return false;
264  }
265  else
266  {
267  if (rollLuck)
268  {
269  this.model.Boulders[x, y].TileOldPosition = this.model.Boulders[x, y].TilePosition;
270  this.model.Boulders[x, y].TilePosition.X -= 1;
271 
272  this.model.Boulders[x - 1, y] = this.model.Boulders[x, y];
273  this.model.Boulders[x, y] = null;
274  }
275  else
276  {
277  return false;
278  }
279  }
280  }
281  }
282 
283  if (dir == Direction.Right)
284  {
285  if (this.model.Boulders[x, y] != null)
286  {
287  if (!this.NothingHere(x + 1, y))
288  {
289  return false;
290  }
291  else
292  {
293  if (rollLuck)
294  {
295  this.model.Boulders[x, y].TileOldPosition = this.model.Boulders[x, y].TilePosition;
296  this.model.Boulders[x, y].TilePosition.X += 1;
297 
298  this.model.Boulders[x + 1, y] = this.model.Boulders[x, y];
299  this.model.Boulders[x, y] = null;
300  }
301  else
302  {
303  return false;
304  }
305  }
306  }
307  }
308 
309  return true;
310  }

793  private void DiamondExpode(int centerX, int centerY)
794  {

796  for (int x = centerX - 1; x < centerX + 2; x++)
797  {
798  for (int y = centerY - 1; y < centerY + 2; y++)
